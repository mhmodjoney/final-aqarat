
DELIMITER $$

CREATE PROCEDURE sp_create_user(
    IN p_email VARCHAR(255),
    IN p_password VARCHAR(255),
    IN p_userName VARCHAR(100),
    IN p_fullName VARCHAR(255),
    IN p_phoneNum VARCHAR(20),
    IN p_whatsappNum VARCHAR(20),
    IN p_type VARCHAR(50),
    IN p_createdBy INT
)
BEGIN
    INSERT INTO user (
        email, password, userName, fullName, phoneNum, whatsappNum, type, createdBy
    ) VALUES (
        p_email, p_password, p_userName, p_fullName, p_phoneNum, p_whatsappNum, p_type, p_createdBy
    );
    SELECT LAST_INSERT_ID() AS userId;
END$$

DELIMITER ;
